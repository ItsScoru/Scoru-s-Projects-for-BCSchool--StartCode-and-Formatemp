CREATE TABLE automobile (
    id INT (10) AUTO_INCREMENT PRIMARY KEY,
    numero_telaio VARCHAR (32) UNIQUE,
    marca VARCHAR (64),
    modello VARCHAR(64),
    id_persona INT (10),
    FOREIGN KEY (id_persona) REFERENCES persona(id)
);
